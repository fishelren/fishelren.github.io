gulp使用简介

发表于 2018-01-02 |

gulp.js是基于Node.js开发的一款自动化工具，作用是使重复的流程化工作按任务的方式自动执行。它使用Node.js的stream来读取和操作数据，速度很快。所有的gulp代码编写都可以看做是将规律转化为代码的过程。

全局安装

npm install -g gulp

全局安装gulp后，还需要在项目文件夹中本地安装一次。全局安装是为了在任意地方使用插件命令，本地安装是为了保证全局插件版本不影响到项目插件版本，且本地安装的模块可以通过require引入，而require引入不了全局安装的模块。

在项目文件夹下执行命令：

npm install --save-dev gulp //将gulp作为项目开发阶段的依赖写入package.json中

将依赖关系写入package.json相当于在package.json中“备案”，当将package.json分享给别人时，对方只需要在命令行npm install就可以检测package.json中记录的依赖并安装所有依赖，而不需要分享node_modules文件夹。

npm install 默认会安装package.json中dependencies字段和devDependencies字段中的所有模块；如果使用–production参数，可以只安装dependencies字段的模块，即仅在发布后需要的依赖模块。

gulpfile.js 是gulp的主文件，gulp 的所有配置代码都写在 gulpfile.js 文件中

var gulp=require("gulp"); //引入局部安装的gulp插件。require() 是 node （CommonJS）中获取模块的语法
gulp.task("default",function(){ //定义名称为default的任务
    console.log("hello world");
});

切换到gulpfile.js所在目录

gulp <任务名> 执行指定任务名的任务，不跟任务名则执行名为"default"的任务

gulp的使用很容易上手，它的学习分为API和插件两个部分（如果要为gulp编写插件另说~）

gulp API

gulp中的常用通配符

  • *      0个或多个字符
  • ?      一个字符
  • […]      匹配方括号中出现字符的任意一个。若以!或^开头表示匹配不出现在方括号中的任意一个字符
  • !(pattern|pattern|pattern)      不匹配出现在圆括号中的任一模式
  • ?(pattern|pattern|pattern)      匹配任一模式0次或1次
  • +(pattern|pattern|pattern)      匹配任一模式1次或多次
  • *(a|b|c)      匹配任一模式0次或多次
  • @(pattern|pat*|pat?erN)      只匹配一次
  • **      多层目录

gulp.src

读取要操作的文件

gulp.src(globs[, options])
    globs：文件匹配模式或其数组，也可以是具体的文件路径
    options：可选参数

eg:

    gulp.src('src/a.js')

    gulp.src('src/**/*.js')

    gulp.src(["js/*.js","!js/*.min.js"])

gulp.dest

将结果导出到文件夹

gulp.dest(path[, options])
    path：写入文件的路径（文件夹，而非文件名）
    options：可选参数

最终导出文件的完整路径为：

gulp.src中没有指定base参数时：

1）有通配符时：path+gulp.src中有通配符开始出现的那部分路径。

eg:
    var gulp = reruire('gulp');
    gulp.src('src/**/*.js')
        .pipe(gulp.dest('dist')); 
        //如果 **/*.js 匹配到的文件为 js/b.js ,则生成的文件路径为 dist/js/b.js

2）没有通配符时：path+gulp.src中的文件名。

eg:
    var gulp = require('gulp');
    gulp.src('src/a.js')
        .pipe(gulp.dest('dist'));
        //最终生成的文件路径为 dist/a.js
        //pipe() 用于传导流，是Node.js的函数

gulp.src中指定base参数时：

3）path+base之后的路径。 

eg:
    gulp.src(src/js/*.js, {base:'src'}) //配置了base参数，此时base路径为src
        //假设匹配到的文件为src/js/c.js
        .pipe(gulp.dest('dist')) //此时生成的文件路径为 dist/js/c.js

gulp.task

定义任务

gulp.task(name[, deps], fn)
    name: 任务名
    deps：当前任务的依赖任务数组，其中的任务会在当前任务执行前执行一遍。可选。
    fn：任务对应的函数。可选。

eg:
    gulp.task('mytask', ['array', 'of', 'task', 'names'], function() {
      // 做一些事
    });

eg:
    //只要执行default任务，就相当于把one,two,three这三个任务执行了
    gulp.task('default',['one','two','three']);

eg:
    gulp.task('one',function(){
      //one是一个异步执行的任务
      setTimeout(function(){
            console.log('one is done')
        },5000);
    });

    //two任务虽然依赖于one任务,但并不会等到one任务中的异步操作完成后再执行
    gulp.task('two',['one'],function(){
        console.log('two is done');
    });

gulp.watch

监听文件的变动。有gulp.watch(glob[, opts], tasks)和gulp.watch(glob[, opts, cb])两种形式

模式1：
gulp.watch(glob[, opts], tasks)
    glob 文件匹配模式或其数组
    opts 为一个可选的配置对象，通常不需要用到
    tasks 文件变动后要执行任务的名称数组

eg:
    gulp.task('uglify',function(){
      //do something
    });
    gulp.task('reload',function(){
      //do something
    });
    gulp.watch('js/**/*.js', ['uglify','reload']);

---    

模式2：
gulp.watch(glob[, opts, cb])
    glob 文件匹配模式或其数组
    opts 为一个可选的配置对象，通常不需要用到
    cb 每次变动需要执行的回调函数

ps：cb会被传入一个event，形式如下：
    event:{
        type:xxx， //event.type表示“ 发生的变动的类型：added, changed 或者 deleted ”
        path:xxx //event.path表示“ 发生变动的文件的路径 ”
    }

eg:
    gulp.watch('js/**/*.js', function(event){
        console.log(event.type); 
        console.log(event.path);
    });

常用插件

gulp-uglify

压缩js文件

eg:
    var gulp=require("gulp");
    var minifyJS=require("gulp-uglify");
    var rename=require("gulp-rename"); //更名插件

    gulp.task("minifyJS",function(){ // 压缩
        gulp.src("src/js/*.js")
            .pipe(minifyJS())
            .pipe(rename({
                suffix:".min" //添加后缀
            }))
            .pipe(gulp.dest("dist/js"))
    });

    gulp.task("monitorJS",function(){ // 监听
        gulp.watch("src/js/*.js",["minifyJS"]);
    });

    gulp.task("default",["minifyJS","monitorJS"]);

以上做法只要有一个文件发生变动就会重新压缩所有的文件，这样在性能上是存在问题的。

利用gulp.watch(glob[, opts, cb])的形式优化性能：

eg:
    var gulp=require("gulp");
    var minifyJS=require("gulp-uglify");
    var rename=require("gulp-rename");
    var gutil=require("gulp-util"); //提供一些简单的工具，这里用到了它的输出功能

    function minifyJSFunc(glob){ // 压缩函数
        gulp.src(glob)
            .pipe(minifyJS())
            .pipe(rename({
                suffix:".min"
            }))
            .pipe(gulp.dest("dist/js"));
    }

    gulp.task("minifyJS",minifyJSFunc("src/js/*.js"));

    gulp.task("monitorJS",function(){
        gulp.watch("src/js/*.js",function(event){
            gutil.log(event.path+" "+event.type); //输出XX文件发生了何种变动
            minifyJSFunc(event.path); //只去重新压缩发生变动的文件
            gutil.log(event.path+" re-compress finished!");
        });
    });

    gulp.task("default",["minifyJS","monitorJS"]);

gulp-clean-css

压缩css文件

npm install --save-dev gulp-clean-css

eg:
    var gulp=require("gulp");
    var minifyCSS=require("gulp-clean-css");
    var rename=require("gulp-rename");
    var gutil=require("gulp-util");

    function minifyCSSFunc(glob){
        gulp.src(glob)
            .pipe(minifyCSS())
            .pipe(rename({
                suffix:".min"
            }))
            .pipe(gulp.dest("dist/css"));
    }

    gulp.task("minifyCSS",minifyCSSFunc("src/css/*.css"));

    gulp.task("monitorCSS",function(){
        gulp.watch("src/css/*.css",function(event){
            gutil.log(event.path+" "+event.type);
            minifyCSSFunc(event.path); //只对发生变动的CSS文件重新压缩
            gutil.log(event.path+" re-compress finished!");
        });
    });

    gulp.task("default",["minifyCSS","monitorCSS"]);

gulp-imagemin

压缩图片

npm install --save-dev gulp-imagemin

eg:
    var gulp=require("gulp");
    var minifyImage=require("gulp-imagemin");

    gulp.task("minifyImage",function(){
        gulp.src("src/images/*.*")
            .pipe(minifyImage())
            .pipe(gulp.dest("dist/images"));
    });

    gulp.task("default",["minifyImage"]);

gulp-zip

打zip包

npm install --save-dev gulp-zip

eg:
    var gulp=require("gulp");
    var zip=require("gulp-zip");

    function formatDatePart(num){ //让日期的某一部分始终为两位数字
        return num<10?"0"+num:""+num;
    }

    gulp.task("default",function(){
        var date=new Date();
        /* // Gulp还不能直接支持ES6（用babel转换另说），而String.prototype.padStart()属于ES8
        var archiveName="archive_"+
            date.getFullYear()+
            (date.getMonth()+1).toString().padStart(2,"0")+
            date.getDate().toString().padStart(2,"0")+
            "_"+
            date.getHours().toString().padStart(2,"0")+
            date.getMinutes().toString().padStart(2,"0")+
            date.getSeconds().toString().padStart(2,"0")+
            ".zip";
        */

        var archiveName="archive_"+
            date.getFullYear()+
            formatDatePart(date.getMonth()+1)+
            formatDatePart(date.getDate())+
            "_"+
            formatDatePart(date.getHours())+
            formatDatePart(date.getMinutes())+
            formatDatePart(date.getSeconds())+
            ".zip";

        gulp.src("src/**/*.*")
            .pipe(zip(archiveName)) //指定zip包的名称
            .pipe(gulp.dest("dist")); //得到的zip包为dist/archiveName
    });
